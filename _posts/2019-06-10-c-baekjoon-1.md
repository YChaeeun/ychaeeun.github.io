## 11650 좌표정렬 (시간초과)

```c
시간초과

#include <stdio.h>
#include <stdlib.h>
#define SWAP(x,y,t) (t=x, x=y, y=t)

int main() {
	int num, i, j, k, keyA, keyB, temp;
	scanf("%d", &num);

	int* listA = (int*)malloc(sizeof(int)*(num+1));
	int* listB = (int*)malloc(sizeof(int)*(num + 1));
	for (i = 0; i < num; i++) {
		scanf("%d %d", &listA[i], &listB[i]);
	}

	for (i = 1; i < num; i++) {
		keyA = listA[i];
		keyB = listB[i];
		j = i - 1;
		while (j >= 0 && listA[j] >= keyA) {
			if (listA[j] == keyA) {
				if (listB[j] < keyB) {
					SWAP(listB[j], keyB, temp);
					continue;
				}
			}
			listA[j + 1] = listA[j];
			listB[j + 1] = listB[j];
			j--;
		}
		listA[j + 1] = keyA;
		listB[j + 1] = keyB;
		
	}
	
	
	for (i = 0; i < num; i++) printf("%d %d\n", listA[i], listB[i]);
	
}

```
